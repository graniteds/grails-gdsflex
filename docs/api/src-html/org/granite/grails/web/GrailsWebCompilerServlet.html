048    public class GrailsWebCompilerServlet extends HttpServlet {
049        
050        private Logger log = Logger.getLogger(GrailsWebCompilerServlet.class);
051    
052        private static final long serialVersionUID = 1L;
053    
054    
055        private ResourceLoader resourceLoader = null;
056        private ResourceLoader servletContextLoader = null;
057        
058        private GrailsApplicationAttributes grailsAttributes;
059        
060        
061        @Override
062        public void init(ServletConfig servletConfig) throws ServletException {
063            this.servletContextLoader = new ServletContextResourceLoader(servletConfig.getServletContext());
064            ApplicationContext springContext = WebApplicationContextUtils.getRequiredWebApplicationContext(servletConfig.getServletContext());
065                    if (springContext.containsBean(GroovyPageResourceLoader.BEAN_ID))
066                            this.resourceLoader = (ResourceLoader)springContext.getBean(GroovyPageResourceLoader.BEAN_ID);
067                    
068            this.grailsAttributes = new DefaultGrailsApplicationAttributes(servletConfig.getServletContext());
069        }
070    
071        @Override
072        protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
073            doPost(request, response);
074        }
075    
076        @Override
077        protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {        
078    
079            request.setAttribute(GrailsApplicationAttributes.REQUEST_SCOPE_ID, grailsAttributes);
080    
081            // Get the name of the Groovy script (intern the name so that we can
082            // lock on it)
083            String pageName = "/swf"+request.getServletPath();        
084            Resource requestedFile = getResourceForUri(pageName);
085            File swfFile = requestedFile.getFile();
086            if (requestedFile == null || swfFile==null || !swfFile.exists()) {
087                response.sendError(404, "\"" + pageName + "\" not found.");
088                return;
089            }
090            response.setContentType("application/x-shockwave-flash");
091            response.setContentLength((int)swfFile.length());
092            response.setBufferSize((int)swfFile.length());
093            response.setDateHeader("Expires", 0);
094            
095            FileInputStream is = null;
096            FileChannel inChan = null;
097            try {
098                is = new FileInputStream(swfFile);
099                OutputStream os = response.getOutputStream();
100                inChan = is.getChannel();
101                long fSize =  inChan.size();
102                MappedByteBuffer mBuf = inChan.map(FileChannel.MapMode.READ_ONLY, 0,fSize);
103                byte[] buf = new byte[(int)fSize];
104                mBuf.get(buf);
105                os.write(buf);
106            } finally {
107                if (is != null)  {
108                    is.close();
109                } 
110                if (inChan != null) {
111                    inChan.close();
112                }
113            }
114        }
115    
116        /**
117         * Attempts to retrieve a reference to a GSP as a Spring Resource instance for the given URI.
118         *
119         * @param uri The URI to check
120         * @return A Resource instance
121         */
122        public Resource getResourceForUri(String uri) {
123            Resource r = getResourceWithinContext(uri);
124            if (r != null && r.exists())
125                return r;
126            
127            // try plugin
128            String pluginUri = GrailsResourceUtils.WEB_INF + uri;
129            r = getResourceWithinContext(pluginUri);
130            if (r != null && r.exists())
131                return r;
132            
133            uri = getUriWithinGrailsViews(uri);
134            return getResourceWithinContext(uri);
135        }
136    
137        private Resource getResourceWithinContext(String uri) {
138            Resource r = servletContextLoader.getResource(uri);
139            if (r.exists()) 
140                return r;
141            return resourceLoader != null ? resourceLoader.getResource(uri) : null;
142        }
143        
144        /**
145         * Returns the path to the view of the relative URI within the Grails views directory
146         *
147         * @param relativeUri The relative URI
148         * @return The path of the URI within the Grails view directory
149         */
150        protected String getUriWithinGrailsViews(String relativeUri) {
151            StringBuffer buf = new StringBuffer();
152            String[] tokens;
153            if (relativeUri.startsWith("/"))
154                relativeUri = relativeUri.substring(1);
155    
156            if (relativeUri.indexOf('/') >= 0)
157                tokens = relativeUri.split("/");
158            else
159                tokens = new String[] { relativeUri };
160    
161            buf.append(GrailsApplicationAttributes.PATH_TO_VIEWS);
162            for (int i = 0; i < tokens.length; i++) {
163                String token = tokens[i];
164                buf.append('/').append(token);
165            }
166            
167            return buf.toString();
168        }
